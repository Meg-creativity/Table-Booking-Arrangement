import React, { useState, useEffect } from 'react';

import {

  Users,

  Clock,

  CheckCircle,

  AlertCircle,

  Plus,

  Trash2,

  Edit3,

  CalendarCheck,

  Timer,

  Ban,

  Calendar as CalendarIcon,

  ChevronLeft,

  ChevronRight,

  X

} from 'lucide-react';

 

const INITIAL_TABLES = [

  { id: '1', capacity: 4 },

  { id: '2', capacity: 4 },

  { id: '3', capacity: 4 },

  { id: '4', capacity: 4 },

  { id: '5', capacity: 4 },

  { id: '81', capacity: 2 },

  { id: '85', capacity: 2 },

];

 

const DINING_DURATION = 90;

const CLEANING_BUFFER = 10; // 清潔緩衝時間 10 分鐘

 

const App = () => {

  const [tables, setTables] = useState(INITIAL_TABLES.map(t => ({ ...t, status: 'idle', currentGuest: null })));

  const getTodayStr = () => new Date().toISOString().split('T')[0];

 

  const [reservations, setReservations] = useState([

    { id: 1, date: getTodayStr(), time: '17:30', endTime: '19:00', nextAvailable: '19:10', name: '王小姐', count: 4, tableId: '1' },

    { id: 2, date: getTodayStr(), time: '19:00', endTime: '20:30', nextAvailable: '20:40', name: '林先生', count: 2, tableId: '81' },

  ]);

 

  const [currentTime, setCurrentTime] = useState(new Date());

  const [view, setView] = useState('tables');

  const [selectedDate, setSelectedDate] = useState(getTodayStr());

 

  const [isCheckInModalOpen, setIsCheckInModalOpen] = useState(false);

  const [isResvModalOpen, setIsResvModalOpen] = useState(false);

  const [editingId, setEditingId] = useState(null);

  const [selectedTable, setSelectedTable] = useState(null);

 

  const [resvForm, setResvForm] = useState({

    date: getTodayStr(),

    time: '11:00',

    name: '',

    count: 1,

    tableId: ''

  });

 

  useEffect(() => {

    const timer = setInterval(() => setCurrentTime(new Date()), 1000);

    return () => clearInterval(timer);

  }, []);

 

  const formatTime = (date) => {

    return date.toLocaleTimeString('zh-TW', { hour: '2-digit', minute: '2-digit', hour12: false });

  };

 

  const timeToMinutes = (timeStr) => {

    const [h, m] = timeStr.split(':').map(Number);

    return h * 60 + m;

  };

 

  const minutesToTime = (totalMinutes) => {

    const h = Math.floor(totalMinutes / 60);

    const m = totalMinutes % 60;

    return `${h.toString().padStart(2, '0')}:${m.toString().padStart(2, '0')}`;

  };

 

  const isBusinessHours = (date) => {

    const timeStr = formatTime(date);

    return (timeStr >= '11:00' && timeStr <= '13:30') || (timeStr >= '17:00' && timeStr <= '19:50');

  };

 

  const getNextResvToday = (tableId) => {

    const todayStr = getTodayStr();

    const nowMin = timeToMinutes(formatTime(currentTime));

    return reservations

      .filter(r => r.tableId === tableId && r.date === todayStr && timeToMinutes(r.time) > nowMin)

      .sort((a, b) => timeToMinutes(a.time) - timeToMinutes(b.time))[0];

  };

 

  const canCheckInNow = (tableId) => {

    const nextResv = getNextResvToday(tableId);

    if (!nextResv) return { can: true };

   

    const nowMin = timeToMinutes(formatTime(currentTime));

    const nextResvMin = timeToMinutes(nextResv.time);

   

    // 扣除清潔時間，計算前一組客人必須離場的時間

    const limitMin = nextResvMin - CLEANING_BUFFER;

    const availableMins = limitMin - nowMin;

 

    if (availableMins < DINING_DURATION) {

      return {

        can: false,

        reason: `準備接預約客 (${nextResv.time})`,

        availableUntil: minutesToTime(limitMin)

      };

    }

    return { can: true, availableUntil: minutesToTime(limitMin) };

  };

 

  const checkTotalConflict = (tableId, dateStr, startTimeStr, excludeId = null) => {

    const startMin = timeToMinutes(startTimeStr);

    const endMin = startMin + DINING_DURATION + CLEANING_BUFFER;

 

    const resvConflict = reservations.some(res => {

      if (res.tableId !== tableId || res.date !== dateStr || res.id === excludeId) return false;

      const resStart = timeToMinutes(res.time);

      const resEnd = timeToMinutes(res.nextAvailable);

      return (startMin < resEnd && resStart < endMin);

    });

    if (resvConflict) return "已預約";

 

    if (dateStr === getTodayStr()) {

      const targetTable = tables.find(t => t.id === tableId);

      if (targetTable && targetTable.status === 'occupied') {

        const guestEndTimeStr = formatTime(targetTable.currentGuest.endTime);

        const guestEndMin = timeToMinutes(guestEndTimeStr) + CLEANING_BUFFER;

        if (startMin < guestEndMin) return "現場用餐中";

      }

    }

    return null;

  };

 

  const handleCheckIn = (tableId, guestCount, name = "現場客人") => {

    const startTime = new Date();

    const endTime = new Date(startTime.getTime() + DINING_DURATION * 60000);

    setTables(prev => prev.map(t => t.id === tableId ? { ...t, status: 'occupied', currentGuest: { name, count: guestCount, startTime, endTime } } : t));

  };

 

  const handleCheckOut = (tableId) => {

    setTables(prev => prev.map(t => t.id === tableId ? { ...t, status: 'idle', currentGuest: null } : t));

  };

 

  const calculateTimes = (startTimeStr) => {

    const [h, m] = startTimeStr.split(':').map(Number);

    const start = new Date();

    start.setHours(h, m, 0);

    const end = new Date(start.getTime() + DINING_DURATION * 60000);

    const next = new Date(end.getTime() + CLEANING_BUFFER * 60000);

    return {

      endTime: formatTime(end),

      nextAvailable: formatTime(next)

    };

  };

 

  const handleSaveResv = (e) => {

    e.preventDefault();

    if (!resvForm.tableId) return;

    const { endTime, nextAvailable } = calculateTimes(resvForm.time);

   

    if (editingId) {

      setReservations(prev => prev.map(r => r.id === editingId ? { ...resvForm, id: editingId, endTime, nextAvailable } : r).sort((a, b) => {

        if (a.date !== b.date) return a.date.localeCompare(b.date);

        return a.time.localeCompare(b.time);

      }));

    } else {

      const newResv = { ...resvForm, id: Date.now(), endTime, nextAvailable };

      setReservations(prev => [...prev, newResv].sort((a, b) => {

        if (a.date !== b.date) return a.date.localeCompare(b.date);

        return a.time.localeCompare(b.time);

      }));

    }

   

    closeModal();

  };

 

  const openEditModal = (res) => {

    setEditingId(res.id);

    setResvForm({

      date: res.date,

      time: res.time,

      name: res.name,

      count: res.count,

      tableId: res.tableId

    });

    setIsResvModalOpen(true);

  };

 

  const closeModal = () => {

    setIsResvModalOpen(false);

    setEditingId(null);

    setResvForm({ date: getTodayStr(), time: '11:00', name: '', count: 1, tableId: '' });

  };

 

  const deleteResv = (id) => setReservations(prev => prev.filter(r => r.id !== id));

 

  const changeSelectedDate = (days) => {

    const current = new Date(selectedDate);

    current.setDate(current.getDate() + days);

    setSelectedDate(current.toISOString().split('T')[0]);

  };

 

  const timeOptions = [];

  for (let h = 11; h <= 20; h++) {

    for (let m = 0; m < 60; m += 10) {

      const time = `${h.toString().padStart(2, '0')}:${m.toString().padStart(2, '0')}`;

      if ((time >= '11:00' && time <= '13:30') || (time >= '17:00' && time <= '19:50')) {

        timeOptions.push(time);

      }

    }

  }

 

  return (

    <div className="min-h-screen bg-slate-50 text-slate-900 pb-10 font-sans">

      <nav className="bg-white border-b px-6 py-4 flex justify-between items-center sticky top-0 z-10 shadow-sm">

        <div className="flex items-center gap-3">

          <div className="bg-orange-500 p-2 rounded-lg text-white"><Users size={24} /></div>

          <div>

            <h1 className="text-xl font-bold tracking-tight">火鍋店現場調度系統</h1>

            <p className="text-xs text-slate-500">{formatTime(currentTime)} | {isBusinessHours(currentTime) ? '🟢 營業中' : '🔴 非接客時段'}</p>

          </div>

        </div>

        <div className="flex bg-slate-100 p-1 rounded-xl shadow-inner">

          <button onClick={() => setView('tables')} className={`px-4 py-2 rounded-lg text-sm font-bold transition-all ${view === 'tables' ? 'bg-white shadow-sm text-orange-600' : 'text-slate-500'}`}>桌位現況</button>

          <button onClick={() => setView('reservations')} className={`px-4 py-2 rounded-lg text-sm font-bold transition-all ${view === 'reservations' ? 'bg-white shadow-sm text-orange-600' : 'text-slate-500'}`}>預約清單</button>

        </div>

      </nav>

 

      <main className="p-6 max-w-7xl mx-auto">

        {view === 'tables' ? (

          <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-4 gap-6">

            {tables.map(table => {

              const isOccupied = table.status === 'occupied';

              const checkInStatus = canCheckInNow(table.id);

              const remaining = isOccupied ? Math.round((table.currentGuest.endTime - currentTime) / 60000) : null;

             

              return (

                <div key={table.id} className={`relative border-2 rounded-3xl p-5 shadow-sm flex flex-col justify-between h-60 transition-all ${

                  isOccupied ? (remaining <= 0 ? 'bg-red-100 border-red-500 text-red-700 animate-pulse' : remaining <= 30 ? 'bg-orange-100 border-orange-500 text-orange-700' : 'bg-green-100 border-green-500 text-green-700') :

                  (!checkInStatus.can ? 'bg-slate-200 border-slate-300 text-slate-400' : 'bg-white border-slate-200 text-slate-500')

                }`}>

                  <div className="flex justify-between items-start">

                    <span className="text-3xl font-black">{table.id}<span className="text-xs ml-1 opacity-60">桌</span></span>

                    <div className="flex flex-col items-end gap-1">

                      <div className="flex items-center gap-1 bg-white/50 px-2 py-1 rounded-full text-xs font-bold border border-black/5"><Users size={12} /> {table.capacity}人</div>

                      {checkInStatus.availableUntil && !isOccupied && (

                        <div className="text-[10px] font-bold text-orange-700 bg-orange-50 px-2 py-0.5 rounded border border-orange-200 animate-in slide-in-from-top-1 duration-300">

                          限用至 {checkInStatus.availableUntil}

                        </div>

                      )}

                    </div>

                  </div>

 

                  {isOccupied ? (

                    <div>

                      <div className="flex items-center gap-2 mb-1 font-bold text-lg truncate text-slate-800"><CheckCircle size={18} className="text-green-500" />{table.currentGuest.name}</div>

                      <div className="flex items-center justify-between mt-4">

                        <div className="text-xs font-bold bg-white/40 px-2 py-1 rounded">剩餘 {remaining} 分鐘</div>

                        <button onClick={() => handleCheckOut(table.id)} className="bg-white px-3 py-1.5 rounded-xl border shadow-sm font-bold text-slate-700 hover:bg-slate-50 text-sm transition-colors">結帳清空</button>

                      </div>

                    </div>

                  ) : (

                    <div className="space-y-3">

                      {!checkInStatus.can ? (

                        <div className="flex flex-col items-center gap-2 opacity-80 py-2">

                          <Ban size={24} className="text-slate-400" />

                          <span className="text-sm font-bold">{checkInStatus.reason}</span>

                        </div>

                      ) : (

                        <div className="flex flex-col gap-2">

                           <div className="text-xs text-center text-slate-400 font-bold mb-1">

                             {checkInStatus.availableUntil ? `⚠️ 下一組預約 ${minutesToTime(timeToMinutes(checkInStatus.availableUntil) + CLEANING_BUFFER)}` : '✨ 目前完全空閒'}

                           </div>

                           <button onClick={() => { setSelectedTable(table); setIsCheckInModalOpen(true); }} className="w-full py-3 rounded-2xl bg-orange-500 text-white font-black hover:bg-orange-600 shadow-lg shadow-orange-100 transition-all text-sm active:scale-95">現場帶位</button>

                        </div>

                      )}

                    </div>

                  )}

                </div>

              );

            })}

          </div>

        ) : (

          <div className="bg-white rounded-3xl shadow-sm border border-slate-200 overflow-hidden">

            <div className="p-6 border-b flex flex-col md:flex-row gap-4 justify-between items-center bg-white">

              <div className="flex items-center gap-4">

                <div className="bg-orange-100 p-2 rounded-xl text-orange-600"><CalendarIcon size={20} /></div>

                <div className="flex items-center gap-2 bg-slate-100 p-1 rounded-xl">

                  <button onClick={() => changeSelectedDate(-1)} className="p-2 hover:bg-white rounded-lg transition-all text-slate-400 hover:text-orange-500"><ChevronLeft size={20} /></button>

                  <span className="px-4 font-black text-slate-700 min-w-[120px] text-center">{selectedDate === getTodayStr() ? '今日 ' : ''}{selectedDate}</span>

                  <button onClick={() => changeSelectedDate(1)} className="p-2 hover:bg-white rounded-lg transition-all text-slate-400 hover:text-orange-500"><ChevronRight size={20} /></button>

                </div>

              </div>

              <button onClick={() => setIsResvModalOpen(true)} className="bg-orange-500 text-white px-5 py-2.5 rounded-2xl text-sm font-bold flex items-center gap-2 hover:bg-orange-600 shadow-lg shadow-orange-100 transition-all">

                <Plus size={18} /> 新增預約

              </button>

            </div>

            <div className="overflow-x-auto">

              <table className="w-full text-left">

                <thead className="bg-slate-50 text-slate-500 text-[10px] font-black uppercase tracking-[0.2em]">

                  <tr>

                    <th className="px-6 py-4">時間</th>

                    <th className="px-6 py-4">結束</th>

                    <th className="px-6 py-4 font-bold text-orange-600">桌號</th>

                    <th className="px-6 py-4">顧客姓名</th>

                    <th className="px-6 py-4">人數</th>

                    <th className="px-6 py-4 text-right">管理</th>

                  </tr>

                </thead>

                <tbody className="divide-y divide-slate-100">

                  {reservations.filter(r => r.date === selectedDate).length === 0 ? (

                    <tr><td colSpan="6" className="px-6 py-16 text-center text-slate-400 italic">該日期尚無預約</td></tr>

                  ) : (

                    reservations.filter(r => r.date === selectedDate).map((res) => (

                      <tr key={res.id} className="hover:bg-slate-50 transition-colors">

                        <td className="px-6 py-4 font-bold text-slate-700">{res.time}</td>

                        <td className="px-6 py-4 text-slate-400 text-sm">{res.endTime}</td>

                        <td className="px-6 py-4 font-black text-xl text-slate-800">{res.tableId}</td>

                        <td className="px-6 py-4 font-bold text-slate-700">{res.name}</td>

                        <td className="px-6 py-4 text-sm text-slate-500">{res.count} 人</td>

                        <td className="px-6 py-4 text-right space-x-2">

                          <button onClick={() => openEditModal(res)} className="text-slate-400 hover:text-blue-500 p-2 transition-colors inline-flex items-center gap-1"><Edit3 size={16} /> <span className="text-xs font-bold">修正</span></button>

                          <button onClick={() => deleteResv(res.id)} className="text-slate-300 hover:text-red-500 p-2 transition-colors"><Trash2 size={16} /></button>

                        </td>

                      </tr>

                    ))

                  )}

                </tbody>

              </table>

            </div>

          </div>

        )}

      </main>

 

      {/* 新增/修正預約 Modal */}

      {isResvModalOpen && (

        <div className="fixed inset-0 bg-black/60 backdrop-blur-sm flex items-center justify-center z-50 p-4">

          <div className="bg-white rounded-[2rem] shadow-2xl w-full max-w-md overflow-hidden animate-in zoom-in duration-300">

            <div className={`p-8 ${editingId ? 'bg-blue-500' : 'bg-orange-500'} text-white flex justify-between items-center relative`}>

              <div className="z-10">

                <h3 className="text-2xl font-black italic uppercase">{editingId ? 'Edit Booking' : 'New Booking'}</h3>

                <p className="text-xs opacity-80 font-bold">{editingId ? `正在修正 ${resvForm.name} 的預約` : '支援跨日登記與智慧防呆'}</p>

              </div>

              <button onClick={closeModal} className="z-10 bg-white/20 p-2 rounded-full hover:bg-white/40 transition-colors"><X size={20} /></button>

            </div>

            <form onSubmit={handleSaveResv} className="p-8 space-y-5">

              <div className="grid grid-cols-2 gap-4">

                <div>

                  <label className="block text-xs font-black text-slate-400 mb-2 uppercase tracking-wider">日期</label>

                  <input type="date" required className="w-full border-2 border-slate-100 rounded-xl p-3 font-bold text-sm outline-none focus:border-orange-500 bg-slate-50" value={resvForm.date} onChange={(e) => setResvForm({...resvForm, date: e.target.value, tableId: ''})} />

                </div>

                <div>

                  <label className="block text-xs font-black text-slate-400 mb-2 uppercase tracking-wider">時間</label>

                  <select className="w-full border-2 border-slate-100 rounded-xl p-3 font-bold text-sm outline-none focus:border-orange-500 bg-slate-50" value={resvForm.time} onChange={(e) => setResvForm({...resvForm, time: e.target.value, tableId: ''})}>

                    {timeOptions.map(t => <option key={t} value={t}>{t}</option>)}

                  </select>

                </div>

              </div>

              <div>

                <label className="block text-xs font-black text-slate-400 mb-2 uppercase tracking-wider">選擇桌號</label>

                <div className="grid grid-cols-2 gap-2 max-h-40 overflow-y-auto pr-1">

                  {INITIAL_TABLES.map(t => {

                    const conflictType = checkTotalConflict(t.id, resvForm.date, resvForm.time, editingId);

                    return (

                      <button

                        key={t.id}

                        type="button"

                        disabled={!!conflictType}

                        onClick={() => setResvForm({...resvForm, tableId: t.id})}

                        className={`p-2.5 rounded-xl border-2 font-bold transition-all flex flex-col items-center justify-center ${

                          resvForm.tableId === t.id

                            ? (editingId ? 'border-blue-500 bg-blue-50 text-blue-600' : 'border-orange-500 bg-orange-50 text-orange-600')

                            : conflictType

                              ? 'bg-slate-100 text-slate-300 border-transparent cursor-not-allowed'

                              : 'border-slate-100 text-slate-700 hover:border-orange-200'

                        }`}

                      >

                        <span className="text-sm">{t.id} 桌</span>

                        {conflictType && <span className="text-[8px] font-normal leading-tight text-red-400">{conflictType === "現場用餐中" ? "現場正有人" : "已預約"}</span>}

                      </button>

                    );

                  })}

                </div>

              </div>

              <div className="grid grid-cols-2 gap-4">

                <input required className="border-2 border-slate-100 rounded-xl p-3 font-bold outline-none focus:border-orange-500 bg-slate-50" type="text" placeholder="顧客姓名" value={resvForm.name} onChange={(e) => setResvForm({...resvForm, name: e.target.value})} />

                <input type="number" min="1" max="4" className="border-2 border-slate-100 rounded-xl p-3 font-bold outline-none focus:border-orange-500 bg-slate-50" value={resvForm.count} onChange={(e) => setResvForm({...resvForm, count: parseInt(e.target.value)})} />

              </div>

              <div className="flex gap-3 pt-2">

                <button type="button" onClick={closeModal} className="flex-1 py-3 font-bold text-slate-400 hover:bg-slate-50 rounded-xl transition-all">取消</button>

                <button type="submit" disabled={!resvForm.tableId} className={`flex-1 py-3 ${editingId ? 'bg-blue-600 shadow-blue-100' : 'bg-orange-500 shadow-orange-100'} text-white font-black rounded-xl shadow-lg disabled:bg-slate-200 disabled:shadow-none active:scale-95 transition-all`}>

                  {editingId ? '確認修正' : '確認登記'}

                </button>

              </div>

            </form>

          </div>

        </div>

      )}

 

      {/* 現場帶位 Modal */}

      {isCheckInModalOpen && (

        <div className="fixed inset-0 bg-black/60 backdrop-blur-sm flex items-center justify-center z-50 p-4">

          <div className="bg-white rounded-[2rem] shadow-2xl w-full max-sm:w-full max-w-sm overflow-hidden animate-in zoom-in duration-200">

            <div className="p-6 bg-slate-800 text-white flex justify-between items-center">

              <h3 className="text-xl font-black italic uppercase text-white">Guest Entry</h3>

              <span className="bg-white/20 px-3 py-1 rounded-full text-xs font-bold">{selectedTable?.id} 號桌</span>

            </div>

            <div className="p-8 space-y-6">

              <input id="guestNameInput" type="text" placeholder="輸入顧客稱呼" className="w-full border-2 border-slate-100 rounded-xl p-4 font-bold text-lg outline-none focus:border-orange-500 bg-slate-50" />

              <div className="flex gap-3">

                <button onClick={() => setIsCheckInModalOpen(false)} className="flex-1 py-4 font-bold text-slate-400 hover:bg-slate-50 rounded-2xl transition-all">取消</button>

                <button onClick={() => {

                  const name = document.getElementById('guestNameInput').value || "現場客人";

                  handleCheckIn(selectedTable.id, 4, name);

                  setIsCheckInModalOpen(false);

                }} className="flex-1 py-4 bg-slate-800 text-white font-black rounded-2xl shadow-xl active:scale-95 transition-all">帶位入座</button>

              </div>

            </div>

          </div>

        </div>

      )}

    </div>

  );

};

 

export default App;
